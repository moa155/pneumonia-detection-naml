\section{Introduction}
%======================================================================

%----------------------------------------------------------------------
\begin{frame}{Motivation}
\begin{columns}
\begin{column}{0.58\textwidth}
\scriptsize
\begin{itemize}\setlength\itemsep{0.1em}
    \item Pneumonia: leading infectious cause of death worldwide
    \item Detection from chest X-rays is challenging:
    \begin{itemize}\setlength\itemsep{0pt}
        \item Subtle opacities, easily missed
        \item Variable sizes and shapes
        \item Overlapping pathologies
    \end{itemize}
    \item Deep learning object detection can localise lesions automatically
    \item \textbf{Question:} do anchor-free detectors (FCOS) truly outperform anchor-based ones for this task, as Wu et al.\ (2024) argue?
\end{itemize}
\end{column}
\begin{column}{0.4\textwidth}
\centering
\includegraphics[width=0.85\textwidth]{detection_samples.png}

\vspace{0.2em}
{\tiny RSNA Pneumonia Detection Challenge\\26{,}684 chest X-rays}
\end{column}
\end{columns}
\end{frame}

%----------------------------------------------------------------------
\begin{frame}{Reference Paper}
\begin{block}{Wu et al., \textit{Scientific Reports} (2024)}
``Pneumonia detection based on RSNA dataset and anchor-free deep learning detector''
\end{block}

\vspace{0.5em}
Key contributions:
\begin{itemize}
    \item First application of FCOS (anchor-free) detector to pneumonia detection
    \item Comparison with anchor-based methods (RetinaNet, Faster R-CNN)
    \item FCOS achieves AP@0.5 $\approx$ 28.5\% on the RSNA dataset
    \item Demonstrates anchor-free advantage for medical object detection
\end{itemize}
\end{frame}

%======================================================================
\section{Background}
%======================================================================

%----------------------------------------------------------------------
\begin{frame}{Detection Paradigms}
\begin{columns}
\begin{column}{0.48\textwidth}
\begin{block}{\textcolor{retinaorange}{Anchor-Based}}
\begin{itemize}
    \item Predefined anchor boxes at each location
    \item Classify anchor as object/background
    \item Regress box offsets from anchor
    \item Need to design anchor sizes \& ratios
\end{itemize}
\textbf{Examples:} RetinaNet, Faster R-CNN
\end{block}
\end{column}
\begin{column}{0.48\textwidth}
\begin{block}{\textcolor{fcosblue}{Anchor-Free (FCOS)}}
\begin{itemize}
    \item Per-pixel classification \& regression
    \item Predict distances $(l, t, r, b)$ to box edges
    \item Center-ness branch for quality
    \item \textbf{No anchor hyperparameters!}
\end{itemize}
\textbf{Advantage:} Simpler, more flexible
\end{block}
\end{column}
\end{columns}
\end{frame}

\section{Experimental Setup}
%======================================================================

%----------------------------------------------------------------------
\begin{frame}{Dataset \& Training Configuration}
\begin{columns}
\begin{column}{0.48\textwidth}
\begin{block}{Dataset}
\begin{itemize}
    \item RSNA Pneumonia Detection Challenge
    \item Full dataset: 26,684 images
    \item Patient-level 80/20 split
    \item DICOM $\to$ PNG preprocessing
    \item $512 \times 512$ resolution
\end{itemize}
\end{block}
\end{column}
\begin{column}{0.48\textwidth}
\begin{block}{Training}
\begin{itemize}
    \item Optimizer: Adam ($\text{lr}=10^{-3}$); SGD for FCOS paper-recipe ablation
    \item Weight decay: $10^{-4}$ (Adam) / $1.6\!\times\!10^{-4}$ (SGD)
    \item Epochs: 40 for every detector
    \item Batch size: 32 / 16 (F-RCNN)
    \item Cosine annealing (Adam) / step decay at $24,32$ (SGD)
    \item EMA (decay = 0.999), backbone frozen first 3 epochs
\end{itemize}
\end{block}
\end{column}
\end{columns}

\vspace{0.5em}
\centering
{\footnotesize \textit{Hardware: 4$\times$ NVIDIA H100 80\,GB SXM (1 model per GPU, parallel training, BF16). End-to-end wall-clock $\approx$ 2.5\,h.}}
\end{frame}

%----------------------------------------------------------------------
\begin{frame}{Data Augmentation}
Medical imaging--specific pipeline (\texttt{albumentations}):
\begin{columns}
\begin{column}{0.48\textwidth}
\begin{itemize}
    \item \textbf{CLAHE} ($p=0.3$) --- local contrast
    \item \textbf{Brightness/Contrast} ($p=0.4$)
    \item \textbf{RandomGamma} ($p=0.2$)
    \item \textbf{HorizontalFlip} ($p=0.5$)
\end{itemize}
\end{column}
\begin{column}{0.48\textwidth}
\begin{itemize}
    \item \textbf{Affine} ($p=0.3$) --- shift/scale/rotate
    \item \textbf{Blur / Noise} ($p=0.2$)
    \item \textbf{Grid/Optical distortion} ($p=0.15$)
    \item \textbf{CoarseDropout} ($p=0.15$)
\end{itemize}
\end{column}
\end{columns}

\vspace{1em}
\begin{itemize}
    \item No vertical flip --- chest X-rays are always upright
    \item All transforms applied jointly to images \textbf{and} bounding boxes
    \item Evaluation: TTA (horizontal flip) + Gaussian Soft-NMS
\end{itemize}
\end{frame}

\textbf{Finding:} val-AP rises through epoch~$24$ ($39.9\%$) then plateaus flat through epoch~$40$ ($39.3\%$). \section{Discussion \& Conclusion}
%======================================================================

%----------------------------------------------------------------------
\begin{frame}{Key Findings}
\footnotesize
\begin{enumerate}\setlength\itemsep{0.3em}
    \item \textbf{All detectors beat the paper's FCOS} ($28.5\%$ AP@0.5):
    FCOS-Adam $35.3$ (+6.8), FCOS-SGD $39.8$ (+11.3), RetinaNet $41.2$ (+12.7), F-RCNN $42.9$ (+14.4).

    \item \textbf{Anchor-based \emph{outperforms} anchor-free in our setup} (counter to Wu et al.)
    \begin{itemize}\setlength\itemsep{0pt}
        \item Same backbone, same recipe, same epochs $\Rightarrow$ anchor priors win on AP@0.5
        \item The paper SGD ablation closes $\sim$half of the gap to anchor-based ($+3.4$) but does not erase it
        \item Paired bootstrap F-RCNN vs RetinaNet: $+1.69$ pts CI $[-0.25,+3.49]$, $p = 0.080$ -- marginal, not significant at $5\%$
    \end{itemize}

    \item \textbf{Ensemble wins the clinically-relevant metric}
    \begin{itemize}\setlength\itemsep{0pt}
        \item WBF (FCOS+RetinaNet): Patient F1 $= 63.0\%$ (+13.5 vs best single)
        \item F-RCNN excluded: its optimal threshold ($0.824$) distorts the 0.3 cut
    \end{itemize}

    \item \textbf{Budget \& reproducibility}: $\sim$2.5\,h on 4$\times$H100, $\sim$\$30 cloud cost; single-command pipeline at \texttt{github.com/moa155/NAML}.
\end{enumerate}
\end{frame}

%----------------------------------------------------------------------
\begin{frame}{Robustness Checks}
We complement the single training run with six post-hoc analyses on the cached predictions ---
no retraining required:
\begin{itemize}
    \item \textbf{Patient-level bootstrap CIs} on AP@0.5 ($B=500$ resamples) $\Rightarrow$ the FCOS-vs.-anchor-based gap is outside the $95\%$ intervals.
    \item \textbf{Three-way split protocol}: Youden threshold fitted on a calibration half, F1 reported on a held-out test half $\Rightarrow$ confirms our F1 numbers are not threshold-on-test artefacts.
    \item \textbf{RSNA-percentile size buckets} (instead of inappropriate COCO 32/96 cutoffs) $\Rightarrow$ shows where each detector excels.
    \item \textbf{FROC + CPM} (the canonical medical-detection metric).
    \item \textbf{Reliability diagrams and ECE} on patient max-scores $\Rightarrow$ quantifies the calibration gap.
    \item \textbf{Learnt aggregator} (logistic regression, 5-fold CV, 7 per-patient features) $\Rightarrow$ replaces the naive ``any box $>\tau$'' rule and lifts FCOS+ensemble patient F1.
\end{itemize}
\centering
\emph{Single-seed retraining, per-site stratification and external-dataset validation remain genuinely future work.}
\end{frame}
